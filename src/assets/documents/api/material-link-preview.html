API reference for Angular Material password strength

npm install --save @angular-material-extensions/password-strength

import {MatPasswordStrengthModule} from '@angular-material-extensions/password-strength;

Directives

MatPasswordStrengthComponent

<mat-password-strength> component.

Selector: mat-progress-bar

Exported as: matPasswordStrength
Properties
Name Description
@Input()

password: string

the password to calculate its strength
@Input()

min: number

the minimum length of the password (needed for validation) | default = 8
@Input()

max: number

the maximum length of the password (needed for validation) | default = 30
@Input()

enableLengthRule: boolean

whether the password's length should be validated | default = true
@Input()

enableLowerCaseLetterRule: boolean

whether the password should contain a lower case letter | default = true

@Input()

enableUpperCaseLetterRule: boolean

whether the password should contain an upper case letter | default = true
@Input()

enableDigitRule: boolean

whether the password should a digit char | default = true
@Input()

enableSpecialCharRule: boolean

whether the password should a special char | default = true
@Input()

externalError: boolean

used to change the color of the password to warn if an external error occurs
@Input()

validators: Criteria[]

user to define the validation rules (see the default values in the src files)
@Output()

onStrengthChanged: EventEmitter <number>

emits the strength of the provided password in % e.g: 20%, 40%, 60%, 80% or 100%
